函数使用三步骤：1、函数原型声明；2、函数体编写；3、函数调用。
```c
#include <stdio.h>
// 指定函数返回值、函数名、函数形参；void表示没有，没有返回值、没有形参
void show(void); 
int main(){
	// 函数声明也可以放在main函数内：void show(void);
    show();
    show();
	return 0;
}
void show(void){
    int count;
    for(count=1;count <= 10;count++) putchar('*');
    putchar('\n');
}
```
# 有参函数
有参数的函数：
```c
#include <stdio.h>
void show(char c, int num);  // 形参名是假名，定义时可与其不一样
// 有参函数声明也可以不声明形参名，到定义的时候再指定形参名
// void show(char, int);
int main(){
    show('&',12);
	return 0;
}
void show(char c, int num) {
    int count;
    for(count=1;count <= num;count++) putchar(c);
    putchar('\n');

}
```
已经废弃的但需要了解一下的：
```c
#include <stdio.h>
void old(n1,n2);
// 形参可以在声明后再声明
int n1,n2;
int main(){
    old(12,12);
	return 0;
}
}
void old(n1,n2){
    printf("%d",n1 * n2);
}

```
对于较小的函数，常不单独声明，而是直接将定义放在第一次调用函数前。
```c
#include <stdio.h>
void show(char c, int num) {
    int count;
    for(count=1;count <= num;count++) putchar(c);
    putchar('\n');
}
int main(){
    show('&',12);
    return 0;
}
```
